PREFIX tpl: <http://tpl.rdlfacade.org/data#>
PREFIX rdf: <http://www.w3.org/1999/02/22-rdf-syntax-ns#> 
PREFIX rdfs: <http://www.w3.org/2000/01/rdf-schema#>
PREFIX p8: <http://standards.tc184-sc4.org/iso/15926/-8/template-model#>
PREFIX p7tm: <http://standards.iso.org/iso/ts/15926/-8/ed-1/tech/reference-data/p7tm#>
SELECT ?qualifies ?name ?description 
{
  tpl:param1 rdfs:subClassOf p7tm:RDLTemplateStatement ;  
                                rdfs:label ?name ;
                                rdfs:subClassOf ?qualifies .
  OPTIONAL { tpl:param1 rdfs:comment ?description . }
  FILTER (?qualifies != p7tm:RDLTemplateStatement && !isBlank(?qualifies)) 
}
