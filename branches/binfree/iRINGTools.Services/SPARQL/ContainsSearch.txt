PREFIX tpl: <http://tpl.rdlfacade.org/data#>
PREFIX rdfs: <http://www.w3.org/2000/01/rdf-schema#>
PREFIX p8: <http://standards.tc184-sc4.org/iso/15926/-8/template-model#>
SELECT ?uri ?label
{ 
  ?uri rdfs:label ?label .
  OPTIONAL {?role a p8:TemplateRoleDescription . FILTER(?role = ?uri)}
  FILTER (isLiteral(?label) && regex(str(?label),'param1','i') && !bound(?role))
}
ORDER BY ?label
