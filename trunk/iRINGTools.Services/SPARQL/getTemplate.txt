PREFIX tpl: <http://tpl.rdlfacade.org/data#>
PREFIX rdf: <http://www.w3.org/1999/02/22-rdf-syntax-ns#> 
PREFIX rdfs: <http://www.w3.org/2000/01/rdf-schema#>

SELECT	?label ?specialization ?defintion
WHERE
{ 
	OPTIONAL	{ tpl:param1	rdf:type	tpl:Template }
	?templateDescription	tpl:hasTemplate	tpl:param1 . 
	tpl:param1				rdfs:label		?label ; 
							rdfs:subClassOf	?specialization ; 
							rdfs:comment	?definition ; 
}
